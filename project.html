#include <stdio.h>

void drawRectangle(int w, int h) {
    for (int i = 0; i < h; i++) {
        for (int j = 0; j < w; j++) {
            if (i == 0 || i == h-1 || j == 0 || j == w-1)
                printf("x");
            else
                printf(" ");
        }
        printf("\n");
    }
}

#include <stdio.h>

void setColor(int r, int g, int b) {
    printf("\033[38;2;%d;%d;%dm", r, g, b);
}

int main() {
    setColor(255, 100, 50); // Orange
    printf("Hello ANSI World!\n");
    printf("\033[0m"); // Reset
    return 0;
}

#include <string.h>
#include <stdio.h>

void searchFile(char *fname, char *str) {
    FILE *fp = fopen(fname, "r");
    char line[512];
    
    while (fgets(line, 512, fp)) {
        if (strstr(line, str)) {
            printf("Found: %s", line);
        }
    }
    fclose(fp);
}

turtle.c 
typedef struct {
    int x, y;
    int angle;
} Turtle;

void moveTurtle(Turtle *t, int steps) {
    // Simplified movement logic
    if (t->angle == 0) t->x += steps;
    if (t->angle == 90) t->y -= steps;
    
    // Toroidal wrap (screen borders)
    if (t->x > 100) t->x = 0;
}
